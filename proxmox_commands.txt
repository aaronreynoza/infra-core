USER="actions-runner@pve"
TOKEN_NAME="ghactions"

# create token (prints the secret once)
pveum user token add "$USER" "$TOKEN_NAME" --privsep 1 --output-format json



# sanity: see the token exists
pveum user token list actions-runner@pve

# grant ACLs to the token principal (user@realm!tokenid)
pveum aclmod / -token 'actions-runner@pve!ghactions' -role PVEAuditor
pveum aclmod /nodes/pve -token 'actions-runner@pve!ghactions' -role PVEVMAdmin
pveum aclmod /storage/local -token 'actions-runner@pve!ghactions' -role PVEDatastoreAdmin
pveum aclmod /storage/local-lvm -token 'actions-runner@pve!ghactions' -role PVEDatastoreAdmin

# verify the ACLs are there
pveum acl list | grep 'actions-runner@pve!ghactions'
